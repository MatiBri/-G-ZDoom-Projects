script 1 (void) { //Monster Death Event
	if(ActivatorTID() == 1337) //Si el ActivatorTID es igual a 1337
	{
		PrintBold(s: "El jugador mato al Former Human"); //Muestra este mensaje
	}
	else if (ActivatorTID() == 5)
	{
		PrintBold(s: "El Pinky mato al Former Human");
	}

}

script 2 ENTER{ // Initialization
	//En lo scripts ENTER, el jugador es el activador del script

	//Cuando el jugador entre al nivel, se le dara un ID
	Thing_ChangeTID(0, 1337);
	//0 es el ID viejo, y el otro sera el nuevo ID que se le dara


	//Cuando el jugador entre al nivel, se le asignara el ID en vez de 0(por defecto) al ID 1337
}
